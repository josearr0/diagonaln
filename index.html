LEER UN NUMERO N, ASIGNAR N A DIAGONAL DE UNA MATRIZ 5X5. ASIGNAR N+1 ARRIBA DIAGNONAL. ASIGNAR N-1 ABAJO DE DIAGONAL


import java.util.Scanner;
public class diagonalN {
public static void main(String[] args) {
int n;
int rodriguez[][];
rodriguez= new int [5][5];
Scanner sc= new Scanner(System.in);

System.out.println("Ingrese un numero: ");
n= sc.nextInt();

for (int i = 0; i < 5; i++) {
for (int j = 0; j < 5; j++) {

if(i==j){
rodriguez[i][j]=n;
}
if(i>j){
rodriguez[i][j]=n-1;
}
if (i<j) {
rodriguez[i][j]=n+1;
}
}
}
System.out.println("------------");
System.out.println("Matriz: ");
System.out.println("");
for (int i = 0; i < 5; i++) {
for (int j = 0; j < 5; j++) {
System.out.print(rodriguez[i][j]+" ");
}
System.out.println(" ");
}
}

}
